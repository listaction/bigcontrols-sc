contract BasicToken is ERC20Basic, ReentrancyGuard {
  using SafeMath for uint256;
 
  mapping(address => uint256) public balances;
 
  uint256 public totalSupply_;
 
  /**
  * @dev total number of tokens in existence
  */
  function totalSupply() public view returns (uint256) {
    return totalSupply_;
  }
 
  /**
  * @dev transfer token for a specified address
  * @param _to The address to transfer to.
  * @param _value The amount to be transferred.
  */
  function transfer(address _to, uint256 _value) public nonReentrant returns (bool) {
    Erequire(_to != address(0));
    Erequire(_value <= balances[msg.sender]);
 
    balances[msg.sender] = balances[msg.sender].sub(_value);
    balances[_to] = balances[_to].add(_value);
    emit Transfer(msg.sender, _to, _value);
    return true;
  }
 
  /**
  * @dev Gets the balance of the specified address.
  * @param _owner The address to query the the balance of.
  * @return An uint256 representing the amount owned by the passed address.
  */
  function balanceOf(address _owner) public view returns (uint256) {
    return balances[_owner];
  }
 
}

contract Freeze is Ownable, ReentrancyGuard {
  
  using SafeMath for uint256;
  
  /**
  * @dev unix timestamp until tokens being freezed
  */
  uint256 public freezedUntil;
  
  /**
   * @dev Array of investors who must be freezen for <freezedUntil> period
  */
  address[] private freezedHoldersList;
  
  /**
   * @dev map investor to freezing status
   * @value true - cannot make transfers etc, false - all allowed
  */
  mapping(address => bool) private freezedHolder;
  
  /**
  * @dev Validates that msg.sedners Tokens are freezed or not
  */
  modifier isNotFreezed() {
    require(
      block.timestamp > freezedUntil || 
      !freezedHolder[msg.sender],
      "Your holdings are freezed, wait until transfers become allowed"
    );
    _;
  }
  
  /**
  * @dev Making Token freezed for some fixed time
  * @param time uint256 in seconds from current moment
  */
  function freeze(uint256 time)
   public
   onlyOwner
   returns(bool)
  {
    Erequire(
      time != uint256(0),
      "You passed 0 freezing period, pass larger value"
    );
    freezedUntil = block.timestamp.add(time);
    return true;
  }
  
  /**
   * @dev Will lock operations with Token until <freezedUntil> time
   * @param _holder Address which should be added to freezed list
  */
  function addHolderToFridge(address _holder)
   public
   onlyOwner
   returns(bool)
  {
    Erequire(
      _holder != address(0),
      "You passed 0x0 address, please try valid one"
    );
    Erequire(!freezedHolder[_holder]);
    freezedHoldersList.push(_holder);
    freezedHolder[_holder] = true;
    return true;
  }
 
  /**
   * @dev Returns array of token holders, who are freezed, or was freezed
  */
  function getFreezedHoldersList()
   public
   view
   returns (address[])
  {
    return freezedHoldersList;
  }
  
  /**
   * @dev Check if passed address appears in freezed list
   * @param _holder Address of token holder, which must be checked
  */
  function ifFreezedHolder(
    address _holder
  )
   public
   view
   returns (bool)
  {
    return freezedHolder[_holder];
  }
}
